#include <stdio.h>
#include <string.h>
#include <time.h>

#define MAX_ITEMS 5
#define MAX_CART 100

typedef struct {
    int no;
    char nama[50];
    int harga;
} Barang;

typedef struct {
    int no;
    int jumlah;
    char nama[50];
    int harga;
    int total;
    int diskon;
} Keranjang;

Barang daftar_barang[MAX_ITEMS] = {
    {1, "Buku Tulis", 10000},
    {2, "Pensil", 5000},
    {3, "Penghapus", 3000},
    {4, "Pulpen", 7000},
    {5, "Penggaris", 8000}
};

Keranjang keranjang[MAX_CART];
int jumlah_barang = 0;

void tampilkanDaftarBarang() {
    printf("\n===================================\n");
    printf("           DAFTAR BARANG           \n");
    printf("===================================\n");
    printf("No  Nama Barang        Harga\n");
    printf("-----------------------------------\n");
    for (int i = 0; i < MAX_ITEMS; i++) {
        printf("%2d  %-15s  Rp %d\n", daftar_barang[i].no, daftar_barang[i].nama, daftar_barang[i].harga);
    }
    printf("===================================\n");
    printf("99. Struk Pembayaran\n");
    printf("77. Reset Pilihan\n");
    printf("00. Keluar\n");
    printf("===================================\n");
}

void tambahKeKeranjang(int no, int jumlah) {
    for (int i = 0; i < MAX_ITEMS; i++) {
        if (daftar_barang[i].no == no) {
            keranjang[jumlah_barang].no = no;
            keranjang[jumlah_barang].jumlah = jumlah;
            strcpy(keranjang[jumlah_barang].nama, daftar_barang[i].nama);
            keranjang[jumlah_barang].harga = daftar_barang[i].harga;
            keranjang[jumlah_barang].total = jumlah * daftar_barang[i].harga;
            keranjang[jumlah_barang].diskon = (jumlah >= 5) ? keranjang[jumlah_barang].total * 0.1 : 0;
            jumlah_barang++;
            break;
        }
    }
}

void cetakStruk() {
    int totalHarga = 0, totalDiskon = 0, totalBayar = 0;

    // Sorting berdasarkan harga dari kecil ke besar
    for (int i = 0; i < jumlah_barang - 1; i++) {
        for (int j = 0; j < jumlah_barang - i - 1; j++) {
            if (keranjang[j].harga > keranjang[j + 1].harga) {
                Keranjang temp = keranjang[j];
                keranjang[j] = keranjang[j + 1];
                keranjang[j + 1] = temp;
            }
        }
    }

    printf("\n===================================\n");
    printf("         STRUK PEMBAYARAN         \n");
    printf("===================================\n");
    printf("No  Jumlah  Nama Barang       Harga     Total     Diskon\n");
    printf("-----------------------------------\n");

    for (int i = 0; i < jumlah_barang; i++) {
        printf("%2d   %5d   %-15s  Rp %-7d  Rp %-7d  Rp %-7d\n",
            keranjang[i].no, keranjang[i].jumlah, keranjang[i].nama, keranjang[i].harga,
            keranjang[i].total, keranjang[i].diskon);
        totalHarga += keranjang[i].total;
        totalDiskon += keranjang[i].diskon;
    }
    totalBayar = totalHarga - totalDiskon;
    printf("===================================\n");
    printf("Total Harga : Rp %d\n", totalHarga);
    printf("Total Diskon: Rp %d\n", totalDiskon);
    printf("Total Bayar : Rp %d\n", totalBayar);
    printf("===================================\n");

    int uang, kembalian;
    do {
        printf("Masukkan Uang: Rp ");
        scanf("%d", &uang);
        if (uang < totalBayar) {
            printf("Uang yang Anda masukkan kurang!\n");
        }
    } while (uang < totalBayar);

    kembalian = uang - totalBayar;
    printf("Kembalian: Rp %d\n", kembalian);
    printf("===================================\n");

    time_t t;
    struct tm *tm_info;
    char buffer[25];

    time(&t);
    tm_info = localtime(&t);
    strftime(buffer, 25, "%Y-%m-%d %H:%M:%S", tm_info);
    printf("Struk dicetak pada: %s\n", buffer);
    printf("===================================\n");
}

void resetKeranjang() {
    jumlah_barang = 0;
    printf("\n===================================\n");
    printf("     KERANJANG TELAH DIRESET!      \n");
    printf("===================================\n");
}

int main() {
    int pilihan, jumlah;

    while (1) {
        tampilkanDaftarBarang();
        printf("Masukkan pilihan: ");
        scanf("%d", &pilihan);

        if (pilihan == 00) {
            printf("Terima kasih telah berbelanja!\n");
            break;
        } else if (pilihan == 99) {
            cetakStruk();
        } else if (pilihan == 77) {
            resetKeranjang();
        } else {
            printf("Masukkan jumlah barang: ");
            scanf("%d", &jumlah);
            tambahKeKeranjang(pilihan, jumlah);
        }
    }

    return 0;
}
